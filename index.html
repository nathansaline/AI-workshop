Inspect the Embedded JavaScript Logic

Click any activity below to view how its system prompt, parameters, and async fetch call are structured in clean JavaScript!

activity_engine.js 
// Select Activity #1 to inspect system instruction configuration:
const activityConfig = {
    id: 1,
    title: "Full Blog Article & Outline Generator",
    systemPrompt: "You are an expert SEO blog author. Write a detailed, engaging blog post...",
    async execute(promptVal, secondaryVal) {
        const response = await fetch("https://generativelanguage.googleapis.com/v1beta/models/gemini-3-flash-preview:generateContent", {
            method: "POST",
            body: JSON.stringify({ contents: [{ parts: [{ text: promptVal }] }] })
        });
        return response.json();
    }
};
